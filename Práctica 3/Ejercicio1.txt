Monitor BaseDeDatos {
	bool libre = 5;
	cond cola; 
	int esperando = 0;

	Procedure llegada() {
		if (libre == 0) {
			esperando ++;
			wait(cola);
		} else 	libre --;
	} 

	Procedure salir() {
		if (esperando > 0) {
			esperando --;
			signal(cola);
		} else libre ++;
	}
}


Process Persona [id: 1..5] {
	Cajero.llegada();
	leer();
	Cajero.fin();
}
